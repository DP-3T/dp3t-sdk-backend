\section{ /v1/exposed }
    \begin{verbatim}
    post /v1/exposed
    \end{verbatim}
Enpoint used to publish the SecretKey.

\subsection{ Request Headers }
\begin{ubparam}{\textwidth}{|H|c|Y|}

\hline

 \\
\hline
\end{ubparam}

\subsection{ Request Body }
\begin{ubparam}{\textwidth}{|H|c|Y|}
\ubheader{Field} & \ubheader{Type} & \ubheader{Description}\\
\hline

 \\
\hline
\end{ubparam}

    \end{verbatim}

 \\

\end{ubparam}

If dayDateStr has the wrong format
